        function applyContextVisuals(div, item) {

            const mode = state.contextMode;

            if (mode === 'focus') {

                // Dim communicative/distracting items

                if (item.label.toLowerCase().includes('slack') || item.label.toLowerCase().includes('teams') || item.label.toLowerCase().includes('email')) {

                    div.style.opacity = '0.3';

                }

            } 

            else if (mode === 'fire') {

                // Pulse critical items

                if (item.type === 'Critical' || item.label.toLowerCase().includes('escalation')) {

                    div.classList.add('pulsing');

                    div.classList.add('border-rose-200');

                }

                // Dim routine items

                if (item.type === 'Routine' || item.type === 'Strategic') {

                    div.style.opacity = '0.4';

                }

            }

        }

        function calculateFriction(w, s) {

            // Heuristic Logic for Demo

            if (w.type === 'Strategic' && s.category === 'Comms') return 'high';

            if (w.type === 'Critical' && s.category === 'Storage') return 'mod';

            if (w.type === 'Routine' && s.category === 'Tracking') return 'low';

            // Random salt

            return Math.random() > 0.85 ? 'high' : (Math.random() > 0.5 ? 'mod' : 'low');

        }


        function getFrictionColor(level) {

            if(level === 'high') return 'bg-rose-400 hover:bg-rose-500';

            if(level === 'mod') return 'bg-amber-300 hover:bg-amber-400';

            return 'bg-emerald-300 hover:bg-emerald-400';

        }

        async function callGemini(prompt) {

            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;

            const payload = { contents: [{ parts: [{ text: prompt }] }] };

            

            try {

                const res = await fetch(url, {

                    method: 'POST',

                    headers: {'Content-Type': 'application/json'},

                    body: JSON.stringify(payload)

                });

                if(!res.ok) throw new Error('API Error');

                const data = await res.json();

                return data.candidates[0].content.parts[0].text;

            } catch (e) {

                console.error(e);

                return "⚠️ AI Service Unavailable. Please check API Key.";

            }

        }

        // --- CHARTS & HELPERS ---

        function findItem(id) {

            return [...ontologyData.workflows, ...ontologyData.systems, ...ontologyData.personas].find(x => x.id === id);

        }


        function getRelatedIds(sourceId) {

            const direct = ontologyData.relationships.filter(r => r.source === sourceId).flatMap(r => r.targets);

            const inverse = ontologyData.relationships.filter(r => r.targets.includes(sourceId)).map(r => r.source);

            return [...new Set([...direct, ...inverse])];

        }


        function isRelated(id1, id2) {

            return getRelatedIds(id1).includes(id2);

        }
